COMPANY MANAGER

Consult companies

QUERY: SELECT id, name FROM company

Delete a company

QUERY: DELETE FROM company WHERE id = ?

Consult company's employees

QUERY: SELECT employee.id, employee.name, employee.email, employee.address, plant.name, plant.country FROM employee INNER JOIN plant on employee.plant_id = plant.id INNER JOIN company on company.id = plant.company_id WHERE company.name = ?

QUERY: INSERT INTO plant (name, address, country, company_id) VALUES (?, ?, ?, ?)

Consult plants

QUERY: SELECT name, address, country, company_id FROM plant

Delete plant

QUERY: DELETE FROM plant WHERE id = ?
